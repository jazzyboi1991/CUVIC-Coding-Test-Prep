# [해설] 네트워크

## 문제 분석
이 문제는 그래프 이론에서 **연결 성분(Connected Components)**의 개수를 찾는 전형적인 문제입니다. 

**핵심 개념**:
- **노드(Node)**: 각각의 컴퓨터
- **간선(Edge)**: 컴퓨터 간의 연결 상태 (`computers[i][j] == 1`)
- **네트워크**: 직접 또는 간접적으로 연결된 노드들의 집합

**핵심 난이도**:
- 모든 컴퓨터를 빠짐없이 확인해야 하며, 이미 확인한 컴퓨터를 중복해서 카운트하지 않도록 '방문 처리'를 관리하는 것이 중요합니다.
- 문제에서 주어진 인접 행렬(`computers`)을 어떻게 효율적으로 순회할 것인지가 관건입니다.

### 핵심 알고리즘 풀이 전략
1. **DFS (깊이 우선 탐색)**:
   - 한 노드에서 시작해 연결된 끝까지 파고듭니다. 
   - 탐색하며 만나는 모든 컴퓨터에 '방문 완료' 도장을 찍습니다. 
   - 더 이상 갈 곳이 없으면 뒤로 돌아와(Backtracking) 다음 탐색을 이어갑니다.
   - 탐색을 새로 시작할 때마다 네트워크 개수를 1씩 증가시킵니다.
2. **BFS (너비 우선 탐색)**:
   - 큐(Queue)를 사용하여 현재 노드와 인접한 모든 노드를 먼저 방문합니다. 
   - DFS와 동일하게 방문 처리를 수행하며 네트워크 개수를 셉니다.
3. **Union-Find (서로소 집합)**:
   - 연결 정보가 주어질 때마다 두 노드의 부모를 합칩니다(Union). 
   - 모든 정보 처리가 끝나면 고유한 부모(Root)의 개수가 곧 네트워크의 개수가 됩니다.
   - (본 해설에서는 가장 직관적인 DFS/BFS 방식을 위주로 설명합니다.)

## 코드 구현

### 1. Python (DFS)

```python
def solution(n, computers):
    answer = 0
    visited = [False] * n
    
    def dfs(v):
        visited[v] = True
        for neighbor in range(n):
            if computers[v][neighbor] == 1 and not visited[neighbor]:
                dfs(neighbor)
                
    for i in range(n):
        if not visited[i]:
            dfs(i)
            answer += 1
            
    return answer
```

```cpp
#include <iostream>

using namespace std;

void dfs(int v, int n, vector<vector<int>>& computers, vector<bool>& visited) {
    visited[v] = true;
    for (int neighbor = 0; neighbor < n; neighbor++) {
        if (computers[v][neighbor] == 1 && !visited[neighbor]) {
            dfs(neighbor, n, computers, visited);
        }
    }
}

int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    vector<bool> visited(n, false);
    
    for (int i = 0; i < n; i++) {

            dfs(i, n, computers, visited);
            answer++;
        }
    }
    
    return answer;
}
```

### 3. Java (DFS)

```java
import java.util.Scanner; // 키보드로부터 입력값을 읽어오기 위한 클래스입니다.

public class Solution {
    // 깊이 우선 탐색(DFS)을 통해 연결된 모든 컴퓨터를 방문합니다.
    public void dfs(int v, int n, int[][] computers, boolean[] visited) {
        visited[v] = true; // 현재 컴퓨터를 '방문 완료'로 표시합니다.
        for (int neighbor = 0; neighbor < n; neighbor++) {
            // 연결되어 있고 아직 방문하지 않은 컴퓨터가 있다면 탐색을 이어갑니다.
            if (computers[v][neighbor] == 1 && !visited[neighbor]) {
                dfs(neighbor, n, computers, visited);
            }
        }
    }

    public int solution(int n, int[][] computers) {
        int answer = 0; // 네트워크의 개수를 저장할 변수입니다.
        boolean[] visited = new boolean[n]; // 각 컴퓨터의 방문 여부를 저장합니다.
        
        for (int i = 0; i < n; i++) {
            // 아직 방문하지 않은 컴퓨터가 있다면 새로운 네트워크를 발견한 것입니다.
            if (!visited[i]) {
                dfs(i, n, computers, visited); // 해당 컴퓨터와 연결된 모든 노드를 방문 처리합니다.
                answer++; // 네트워크 개수를 하나 올립니다.
            }
        }
        
        return answer;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in); 
        sc.useDelimiter("[^0-9]+"); // 숫자 이외의 모든 문자를 무시하여 입력 편의성을 높입니다.
        
        System.out.print("컴퓨터의 개수(n)를 입력하세요: ");
        int n = sc.nextInt();
        int[][] computers = new int[n][n];

        System.out.println("인접 행렬(computers)을 입력하세요 (예: [[1,1,0],...]):");
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                computers[i][j] = sc.nextInt();
            }
        }
        Solution sol = new Solution();
        System.out.println("결과: " + sol.solution(n, computers));
    }
}
```

## BFS 코드 구현

### 1. Python (BFS)

```python
from collections import deque # 효율적인 큐 사용을 위해 deque를 불러옵니다.
import re # 숫자만 골라내는 정규표현식 도구입니다.

def solution(n, computers):
    answer = 0
    visited = [False] * n
    
    for i in range(n):
        if not visited[i]:
            # 새로운 네트워크 발견! BFS 탐색을 시작합니다.
            queue = deque([i])
            visited[i] = True
            
            while queue:
                v = queue.popleft()
                for neighbor in range(n):
                    # 연결되어 있고 아직 방문하지 않은 인접 노드를 큐에 넣습니다.
                    if computers[v][neighbor] == 1 and not visited[neighbor]:
                        visited[neighbor] = True
                        queue.append(neighbor)
            answer += 1 # 네트워크 개수 증가
            
    return answer

if __name__ == "__main__":
    n = int(input("컴퓨터의 개수(n)를 입력하세요: "))
    print("인접 행렬(computers)을 입력하세요 (예: [[1,1,0],...]):")
    
    # 대괄호나 쉼표가 섞인 입력에서도 숫자만 정확히 뽑아냅니다.
    raw_text = ""
    while len(re.findall(r'\d+', raw_text)) < n * n:
        raw_text += " " + input()
    all_nums = list(map(int, re.findall(r'\d+', raw_text)))
    computers = [all_nums[i*n : (i+1)*n] for i in range(n)]
    
    print(f"결과: {solution(n, computers)}")
```

### 2. C++ (BFS)

```cpp
#include <iostream>
#include <vector>
#include <queue> // BFS를 위한 큐 라이브러리입니다.

using namespace std;

int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    vector<bool> visited(n, false);
    
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            // BFS 탐색을 위한 큐를 생성합니다.
            queue<int> q;
            q.push(i);
            visited[i] = true;
            
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                
                for (int neighbor = 0; neighbor < n; neighbor++) {
                    if (computers[v][neighbor] == 1 && !visited[neighbor]) {
                        visited[neighbor] = true;
                        q.push(neighbor);
                    }
                }
            }
            answer++; // 하나의 연결된 덩어리 탐색 완료
        }
    }
    return answer;
}

int main() {
    int n;
    cout << "컴퓨터의 개수(n)를 입력하세요: ";
    cin >> n;
    vector<vector<int>> computers(n, vector<int>(n));
    cout << "인접 행렬을 입력하세요 (예: [[1,1,0],...]):" << endl;
    
    for(int i=0; i<n; i++) {
        for(int j=0; j<n; j++) {
            // 숫자가 나올 때까지 특수문자를 건너뛰는 견고한 입력 방식입니다.
            while (cin.peek() != EOF && (cin.peek() < '0' || cin.peek() > '9')) cin.ignore();
            cin >> computers[i][j];
        }
    }
    cout << "결과: " << solution(n, computers) << endl;
    return 0;
}
```

### 3. Java (BFS)

```java
import java.util.LinkedList; // 큐를 구현하기 위해 연결 리스트를 사용합니다.
import java.util.Queue;      // 큐 인터페이스입니다.
import java.util.Scanner;    // 입력 도구입니다.

public class Solution {
    public int solution(int n, int[][] computers) {
        int answer = 0;
        boolean[] visited = new boolean[n];
        
        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                // BFS를 위해 큐를 준비하고 시작 노드를 넣습니다.
                Queue<Integer> queue = new LinkedList<>();
                queue.offer(i);
                visited[i] = true;
                
                while (!queue.isEmpty()) {
                    int v = queue.poll();
                    for (int neighbor = 0; neighbor < n; neighbor++) {
                        if (computers[v][neighbor] == 1 && !visited[neighbor]) {
                            visited[neighbor] = true;
                            queue.offer(neighbor);
                        }
                    }
                }
                answer++;
            }
        }
        return answer;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        sc.useDelimiter("[^0-9]+"); // 숫자 이외의 문자를 무시하여 대괄호 형식도 읽을 수 있게 합니다.
        
        System.out.print("컴퓨터의 개수(n)를 입력하세요: ");
        int n = sc.nextInt();
        int[][] computers = new int[n][n];

        System.out.println("인접 행렬을 입력하세요:");
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                computers[i][j] = sc.nextInt();
            }
        }
        Solution sol = new Solution();
        System.out.println("결과: " + sol.solution(n, computers));
    }
}
```

## Union-Find (서로소 집합) 코드 구현

### 1. Python (Union-Find)

```python
import re

def solution(n, computers):
    # 각 컴퓨터의 부모를 자기 자신으로 초기화합니다.
    parent = [i for i in range(n)]
    
    # 해당 컴퓨터가 속한 네트워크의 '대표(루트)'를 찾는 함수입니다.
    def find(parent, x):
        if parent[x] == x:
            return x
        # 경로 압축: 찾는 과정에서 부모를 루트로 바로 연결해 효율을 높입니다.
        parent[x] = find(parent, parent[x])
        return parent[x]
        
    # 두 컴퓨터를 하나의 네트워크로 합치는 함수입니다.
    def union(parent, x, y):
        rootX = find(parent, x)
        rootY = find(parent, y)
        if rootX != rootY:
            parent[rootY] = rootX # 다른 네트워크라면 하나로 합칩니다.
            
    # 모든 연결 정보를 확인하며 네트워크를 합칩니다.
    for i in range(n):
        for j in range(i + 1, n): # 대칭이므로 절반만 확인해도 됩니다.
            if computers[i][j] == 1:
                union(parent, i, j)
                
    # 최종적으로 서로 다른 루트(대표)의 개수를 세면 네트워크의 개수가 됩니다.
    return len(set(find(parent, i) for i in range(n)))

if __name__ == "__main__":
    n = int(input("컴퓨터의 개수(n)를 입력하세요: "))
    print("인접 행렬을 입력하세요:")
    raw_text = ""
    while len(re.findall(r'\d+', raw_text)) < n * n:
        raw_text += " " + input()
    all_nums = list(map(int, re.findall(r'\d+', raw_text)))
    computers = [all_nums[i*n : (i+1)*n] for i in range(n)]
    print(f"결과: {solution(n, computers)}")
```

### 2. C++ (Union-Find)

```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

// 부모를 찾는 find 함수 (경로 압축 기술 적용)
int findParent(vector<int>& parent, int x) {
    if (parent[x] == x) return x;
    return parent[x] = findParent(parent, parent[x]);
}

// 두 집합을 합치는 union 함수
void unionParent(vector<int>& parent, int x, int y) {
    int rootX = findParent(parent, x);
    int rootY = findParent(parent, y);
    if (rootX != rootY) parent[rootY] = rootX;
}

int solution(int n, vector<vector<int>> computers) {
    vector<int> parent(n);
    for (int i = 0; i < n; i++) parent[i] = i; // 초기화

    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (computers[i][j] == 1) {
                unionParent(parent, i, j);
            }
        }
    }

    set<int> roots;
    for (int i = 0; i < n; i++) {
        roots.insert(findParent(parent, i)); // 고유한 루트의 개수를 셉니다.
    }
    return roots.size();
}

int main() {
    int n;
    cout << "컴퓨터의 개수(n)를 입력하세요: ";
    cin >> n;
    vector<vector<int>> computers(n, vector<int>(n));
    cout << "인접 행렬을 입력하세요:" << endl;
    for(int i=0; i<n; i++) {
        for(int j=0; j<n; j++) {
            while (cin.peek() != EOF && (cin.peek() < '0' || cin.peek() > '9')) cin.ignore();
            cin >> computers[i][j];
        }
    }
```

### 3. Java (Union-Find)

```java
import java.util.HashSet;
import java.util.Scanner;
import java.util.Set;

public class Solution {
    // 부모 노드를 찾는 함수 (경로 압축 기술 사용)
    public int find(int[] parent, int x) {
        if (parent[x] == x) return x;
        return parent[x] = find(parent, parent[x]);
    }

    // 두 노드를 하나의 네트워크로 묶는 함수
    public void union(int[] parent, int x, int y) {
        int rootX = find(parent, x);
        int rootY = find(parent, y);
        if (rootX != rootY) {
            parent[rootY] = rootX;
        }
    }

    public int solution(int n, int[][] computers) {
        int[] parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = i; // 자기 자신을 부모로 초기화

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (computers[i][j] == 1) {
                    union(parent, i, j); // 연결된 경우 합칩니다.
                }
            }
        }

        // 고유한 루트 노드의 개수가 곧 네트워크의 개수입니다.
        Set<Integer> roots = new HashSet<>();
        for (int i = 0; i < n; i++) {
            roots.add(find(parent, i));
        }
        return roots.size();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        sc.useDelimiter("[^0-9]+");
        
        System.out.print("컴퓨터의 개수(n)를 입력하세요: ");
        int n = sc.nextInt();
        int[][] computers = new int[n][n];

        System.out.println("인접 행렬을 입력하세요:");
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                computers[i][j] = sc.nextInt();
            }
        }
        Solution sol = new Solution();
        System.out.println("결과: " + sol.solution(n, computers));
    }
}
```

## 핵심 포인트
- **인접 행렬 활용**: 문제에서 주어지는 `computers` 배열 자체가 인접 행렬이므로 별도의 그래프 변환 과정이 필요 없습니다.
- **방문 처리**: 중복 탐색을 막기 위한 `visited` 배열 관리가 필수적입니다. (DFS/BFS의 경우)
- **네트워크 개수**: 탐색이 끊길 때마다 `answer`를 증가시키거나(DFS/BFS), 최종적인 루트의 개수를 세는 것(Union-Find)이 핵심입니다.
