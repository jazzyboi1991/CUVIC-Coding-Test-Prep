# [DFS/BFS] 단어 변환 - 해설

## 1. 문제 분석
이 문제는 한 번에 한 글자만 바꿀 수 있다는 제약 조건을 지키면서, 시작 단어에서 목표 단어까지 가는 **최단 경로**를 찾는 문제입니다.

- **알고리즘 유형**: 너비 우선 탐색 (BFS)
- **핵심 포인트**: 
  - 각 단어를 그래프의 '노드(Node)'로 생각하고, 한 글자만 차이 나는 단어들끼리 '간선(Edge)'으로 연결되어 있다고 가정합니다.
  - 가중치가 없는 그래프에서 최단 거리를 찾는 문제이므로 **BFS**가 가장 적합합니다.
  - 만약 `target` 단어가 `words` 목록에 없다면 절대로 도달할 수 없으므로 즉시 0을 반환해야 합니다.

## 2. 단계별 풀이 전략

1. **사전 체크**:
   - `target`이 `words` 리스트 안에 있는지 확인합니다. 없으면 0을 반환합니다.

2. **단어 비교 로직**:
   - 두 단어를 입력받아 서로 다른 글자가 정확히 **1개**인 경우에만 `true`를 반환하는 함수를 만듭니다.

3. **BFS 탐색**:
   - 큐(Queue)를 생성하고 시작 단어와 현재 단계 수(0)를 넣습니다. `(begin, 0)`
   - 방문 여부를 확인하기 위한 `visited` 배열을 준비합니다.
   - 큐에서 단어를 하나씩 꺼내며:
     - 꺼낸 단어가 `target`과 같다면 현재까지의 단계 수를 반환합니다.
     - 아직 방문하지 않은 단어들 중에서 현재 단어와 한 글자만 다른 단어들을 찾아 큐에 넣고 방문 표시를 합니다.

4. **결과 반환**:
   - 큐가 빌 때까지 `target`을 찾지 못하면 변환이 불가능한 것이므로 0을 반환합니다.

## 3. 코드 구현

### Python
```python
from collections import deque # BFS를 위한 큐 자료구조를 불러옵니다.

def can_convert(word1, word2):
    """두 단어가 한 글자만 다른지 확인하는 함수입니다."""
    diff_count = 0
    for w1, w2 in zip(word1, word2):
        if w1 != w2:
            diff_count += 1
    return diff_count == 1

def solution(begin, target, words):
    # 목표 단어가 목록에 없으면 바로 0을 반환합니다.
    if target not in words:
        return 0
    
    # (현재 단어, 단계 수)를 큐에 넣고 시작합니다.
    queue = deque([(begin, 0)])
    # 방문한 단어를 기록하여 무한 루프를 방지합니다.
    visited = [False] * len(words)
    
    while queue:
        curr_word, step = queue.popleft()
        
        # 목표 단어에 도달했다면 현재까지의 단계를 반환합니다.
        if curr_word == target:
            return step
        
        # 목록에 있는 단어들을 하나씩 확인합니다.
        for i in range(len(words)):
            # 아직 방문하지 않았고, 현재 단어에서 변환 가능하다면
            if not visited[i] and can_convert(curr_word, words[i]):
                visited[i] = True # 방문 표시를 하고
                queue.append((words[i], step + 1)) # 큐에 넣습니다.
                
    return 0

if __name__ == "__main__":
    # 사용자로부터 데이터를 입력받습니다.
    begin = input("시작 단어: ")
    target = input("목표 단어: ")
    words = input("단어 목록(공백 구분): ").split()
    
    print(f"최소 변환 단계: {solution(begin, target, words)}")
```

### C++
```cpp
#include <iostream> // 표준 입출력 라이브러리입니다.
#include <vector>   // 가변 배열을 사용하기 위해 필요합니다.
#include <string>   // 문자열 처리를 위해 필요합니다.
#include <queue>    // BFS 탐색을 위한 큐 자료구조입니다.
#include <algorithm> // find 함수 등을 사용하기 위해 불러옵니다.

#define endl "\n"

using namespace std;

/**
 * [canConvert 함수 설명]
 * 두 단어의 글자 차이가 정확히 1개인지 확인합니다.
 */
bool canConvert(string a, string b) {
    int diff = 0;
    for (int i = 0; i < a.length(); i++) {
        if (a[i] != b[i]) diff++;
    }
    return diff == 1;
}

/**
 * [solution 함수 설명]
 * BFS를 사용하여 begin에서 target까지 가는 최단 단계를 찾습니다.
 */
int solution(string begin, string target, vector<string> words) {
    // target이 words 목록에 있는지 확인
    bool exists = false;
    for (string w : words) {
        if (w == target) exists = true;
    }
    if (!exists) return 0;

    // (현재 단어, 단계 수)를 담는 큐
    queue<pair<string, int>> q;
    q.push({begin, 0});

    // 방문 여부 체크
    vector<bool> visited(words.size(), false);

    while (!q.empty()) {
        string curr = q.front().first;
        int step = q.front().second;
        q.pop();

        if (curr == target) return step;

        for (int i = 0; i < words.size(); i++) {
            // 방문하지 않았고 한 글자만 다르다면
            if (!visited[i] && canConvert(curr, words[i])) {
                visited[i] = true;
                q.push({words[i], step + 1});
            }
        }
    }

    return 0;
}

int main() {
    string b, t;
    int n;
    cout << "시작 단어와 목표 단어를 입력하세요: ";
    cin >> b >> t;
    cout << "단어 목록의 개수를 입력하세요: ";
    cin >> n;
    vector<string> w(n);
    cout << "단어들을 입력하세요: " << endl;
    for (int i = 0; i < n; i++) cin >> w[i];

    cout << "최소 단계: " << solution(b, t, w) << endl;
    return 0;
}
```

### Java
```java
import java.util.*; // Queue, LinkedList 등 다양한 자료구조를 위해 사용합니다.

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in); // 키보드 입력을 받기 위한 도구
        
        System.out.print("시작 단어: ");
        String begin = sc.next();
        System.out.print("목표 단어: ");
        String target = sc.next();
        System.out.print("단어 목록 개수: ");
        int n = sc.nextInt();
        String[] words = new String[n];
        System.out.println("단어들을 입력하세요:");
        for (int i = 0; i < n; i++) words[i] = sc.next();
        
        System.out.println("최소 단계: " + solution(begin, target, words));
        sc.close();
    }

    /**
     * 한 글자만 차이 나는지 확인하는 헬퍼 함수
     */
    private static boolean canConvert(String a, String b) {
        int diff = 0;
        for (int i = 0; i < a.length(); i++) {
            if (a.charAt(i) != b.charAt(i)) diff++;
        }
        return diff == 1;
    }

    /**
     * BFS를 이용한 최단 경로 탐색
     */
    public static int solution(String begin, String target, String[] words) {
        // target이 존재하는지 먼저 확인
        boolean exists = false;
        for (String w : words) {
            if (w.equals(target)) {
                exists = true;
                break;
            }
        }
        if (!exists) return 0;

        // BFS를 위한 큐 생성 (단어와 깊이를 함께 저장하기 위해 클래스나 배열 사용)
        Queue<Node> q = new LinkedList<>();
        q.add(new Node(begin, 0));
        
        boolean[] visited = new boolean[words.length];

        while (!q.isEmpty()) {
            Node curr = q.poll();

            if (curr.word.equals(target)) {
                return curr.count;
            }

            for (int i = 0; i < words.length; i++) {
                if (!visited[i] && canConvert(curr.word, words[i])) {
                    visited[i] = true;
                    q.add(new Node(words[i], curr.count + 1));
                }
            }
        }

        return 0;
    }

    // 단어와 단계를 묶어서 저장하기 위한 간단한 클래스
    static class Node {
        String word;
        int count;

        Node(String word, int count) {
            this.word = word;
            this.count = count;
        }
    }
}
```

## 4. 핵심 포인트 및 주의 사항
- **BFS vs DFS**: 최단 거리를 구하는 문제이므로 BFS가 효율적입니다. DFS를 사용할 경우 모든 경로를 탐색하여 그 중 최솟값을 찾아야 하므로 성능이 떨어질 수 있습니다.
- **방문 체크**: `visited` 배열을 사용하여 이미 확인한 단어를 다시 확인하지 않도록 해야 무한 루프에 빠지지 않습니다.
- **예외 상황**: `target`이 목록에 아예 없는 경우를 처음에 걸러주면 불필요한 탐색을 줄일 수 있습니다.
