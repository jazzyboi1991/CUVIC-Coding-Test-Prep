# [LeetCode] 124. Binary Tree Maximum Path Sum - 해설

## 1. 문제 분석
이 문제는 이진 트리 내에서 연결된 노드들의 합이 최대가 되는 경로를 찾는 문제입니다.

- **알고리즘 유형**: 그래프 탐색 (DFS), 트리 동적 계획법 (Tree DP)
- **시간 복잡도**: $O(N)$ - 트리의 모든 노드를 한 번씩 방문합니다.
- **핵심 포인트**: 
  - 어떤 노드 `U`를 지나는 최대 경로는 `U의 값 + 왼쪽 자식에서의 최대 기여도 + 오른쪽 자식에서의 최대 기여도`가 됩니다.
  - 하지만 `U`가 자신의 부모 노드에게 전달할 수 있는 '최대 기여도'는 **왼쪽 혹은 오른쪽 중 한 방향**만 선택한 경로여야 합니다. (단순 경로의 정의 때문)

## 2. 단계별 풀이 전략

1. **후위 순회(Post-order Traversal) 활용**:
   - 자식 노드들의 정보를 먼저 알아야 현재 노드의 최대 경로를 계산할 수 있으므로, 왼쪽 자식 -> 오른쪽 자식 -> 현재 노드 순으로 탐색하는 DFS를 사용합니다.

2. **기여도(Gain) 계산**:
   - `get_max_gain(node)` 함수는 해당 노드에서 아래로 뻗어나갈 수 있는 한 갈래의 최대 경로 합을 반환합니다.
   - 만약 자식 쪽에서 올라온 합이 음수라면, 그쪽은 선택하지 않는 것(0으로 처리)이 이득입니다.
   - `max_gain = node.val + max(left_gain, right_gain)`

3. **전역 최댓값 갱신**:
   - 각 노드마다 `현재 노드 값 + 왼쪽 기여도 + 오른쪽 기여도`를 계산하여 현재까지 발견한 전역 최댓값(`max_sum`)을 계속 업데이트합니다.
   - 이 과정에서 왼쪽과 오른쪽을 모두 연결한 '아치형' 경로가 고려됩니다.

## 3. 코드 구현

### Python
```python
import sys

# 트리의 노드를 정의하는 클래스입니다.
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class Solution:
    def maxPathSum(self, root):
        # 최종 정답을 저장할 변수입니다. 매우 작은 값으로 초기화합니다.
        self.max_sum = float('-inf')

        def get_max_gain(node):
            if not node:
                return 0

            # 1. 왼쪽과 오른쪽 자식으로부터 얻을 수 있는 최대 이득을 구합니다.
            # 0보다 작으면 더하지 않는 것이 이득이므로 max(0, ...)를 취합니다.
            left_gain = max(get_max_gain(node.left), 0)
            right_gain = max(get_max_gain(node.right), 0)

            # 2. 현재 노드를 정점으로 하는 새로운 경로의 합을 구하고 전역 최댓값을 갱신합니다.
            # 이 경로는 왼쪽 자식 -> 현재 노드 -> 오른쪽 자식으로 이어집니다.
            current_path_sum = node.val + left_gain + right_gain
            self.max_sum = max(self.max_sum, current_path_sum)

            # 3. 부모 노드에게는 현재 노드와 두 자식 중 더 큰 쪽 하나만 골라서 전달합니다.
            # 그래야 부모 쪽에서 경로가 끊기지 않고 이어질 수 있기 때문입니다.
            return node.val + max(left_gain, right_gain)

        get_max_gain(root)
        return self.max_sum

# 입력을 트리 구조로 변환하기 위한 보조 함수입니다.
def build_tree(nodes):
    if not nodes: return None
    root = TreeNode(int(nodes[0]))
    queue = [root]
    i = 1
    while queue and i < len(nodes):
        node = queue.pop(0)
        if nodes[i] != "null":
            node.left = TreeNode(int(nodes[i]))
            queue.append(node.left)
        i += 1
        if i < len(nodes) and nodes[i] != "null":
            node.right = TreeNode(int(nodes[i]))
            queue.append(node.right)
        i += 1
    return root

if __name__ == "__main__":
    print("트리 원소들을 레벨 순서대로 입력하세요 (예: 1 2 3 / null은 빈 공간):")
    input_data = sys.stdin.readline().split()
    if input_data:
        root = build_tree(input_data)
        sol = Solution()
        print(f"최대 경로 합: {sol.maxPathSum(root)}")
```

### C++
```cpp
#include <iostream>  // 입출력을 위한 라이브러리입니다.
#include <vector>    // 리스트 관리를 위해 사용합니다.
#include <string>    // 문자열 처리를 위해 필요합니다.
#include <algorithm> // max 함수 사용을 위해 불러옵니다.
#include <climits>   // INT_MIN 값을 사용하기 위해 불러옵니다.
#include <queue>     // 트리 구축을 위한 큐입니다.

using namespace std;

// 이진 트리의 노드 구조체 정의
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

class Solution {
private:
    int max_sum = INT_MIN; // 결과를 담을 변수 (최소값으로 초기화)

public:
    int maxPathSum(TreeNode* root) {
        get_max_gain(root);
        return max_sum;
    }

    // 각 노드에서의 최대 기여도를 계산하는 재귀 함수
    int get_max_gain(TreeNode* node) {
        if (node == NULL) return 0;

        // 1. 왼쪽/오른쪽 자식의 최대 기여도를 구합니다. 
        // 0보다 작은 결과는 무시하여 합을 최대화합니다.
        int left_gain = max(get_max_gain(node->left), 0);
        int right_gain = max(get_max_gain(node->right), 0);

        // 2. 현재 노드를 아치형 경로의 꼭짓점으로 보았을 때의 합을 구합니다.
        int current_path_sum = node->val + left_gain + right_gain;
        // 3. 전체 최대값과 비교하여 더 크면 갱신합니다.
        max_sum = max(max_sum, current_path_sum);

        // 4. 부모 쪽으로는 한 갈래(자신 + 더 큰 자식 한쪽)만 선택하여 전달합니다.
        return node->val + max(left_gain, right_gain);
    }
};

// 입력을 트리 구조로 바꿔주는 도우미 함수
TreeNode* buildTree(vector<string> nodes) {
    if (nodes.empty() || nodes[0] == "null") return NULL;
    TreeNode* root = new TreeNode(stoi(nodes[0]));
    queue<TreeNode*> q;
    q.push(root);
    int i = 1;
    while (!q.empty() && i < nodes.size()) {
        TreeNode* curr = q.front(); q.pop();
        if (nodes[i] != "null") {
            curr->left = new TreeNode(stoi(nodes[i]));
            q.push(curr->left);
        }
        i++;
        if (i < nodes.size() && nodes[i] != "null") {
            curr->right = new TreeNode(stoi(nodes[i]));
            q.push(curr->right);
        }
        i++;
    }
    return root;
}

int main() {
    cout << "트리 노드들을 입력하세요 (예: -10 9 20 null null 15 7):" << endl;
    string val;
    vector<string> nodes;
    while (cin >> val) {
        nodes.push_back(val);
        if (cin.peek() == '\n') break;
    }
    
    TreeNode* root = buildTree(nodes);
    Solution sol;
    cout << "최대 경로 합: " << sol.maxPathSum(root) << endl;
    
    return 0;
}
```

### Java
```java
import java.util.*; // Queue, LinkedList 등을 위해 불러옵니다.

// 트리 노드 클래스 정의
class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode(int x) { val = x; }
}

public class Main {
    private static int maxSum = Integer.MIN_VALUE; // 최종 결과를 담을 전역 변수

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("트리 데이터를 입력하세요 (예: 1 2 3):");
        if (sc.hasNextLine()) {
            String[] nodes = sc.nextLine().split(" ");
            TreeNode root = buildTree(nodes);
            System.out.println("최대 경로 합: " + maxPathSum(root));
        }
        sc.close();
    }

    public static int maxPathSum(TreeNode root) {
        maxSum = Integer.MIN_VALUE; // 초기화
        getMaxGain(root);
        return maxSum;
    }

    /**
     * [getMaxGain 함수 설명]
     * 재귀적으로 자식의 기여도를 확인하고 전역 최댓값을 갱신합니다.
     */
    private static int getMaxGain(TreeNode node) {
        if (node == null) return 0;

        // 1. 자식들이 줄 수 있는 최대 이득을 구합니다. (음수면 버립니다)
        int leftGain = Math.max(getMaxGain(node.left), 0);
        int rightGain = Math.max(getMaxGain(node.right), 0);

        // 2. 현재 노드를 중간 다리로 쓰는 경로의 총합을 구합니다.
        int currentPathSum = node.val + leftGain + rightGain;
        
        // 3. 지금까지의 최고 기록보다 크면 갈아치웁니다.
        maxSum = Math.max(maxSum, currentPathSum);

        // 4. 부모 노드 쪽으로는 '나 + 더 나은 자식 한 명'만 데려가서 연장합니다.
        return node.val + Math.max(leftGain, rightGain);
    }

    // 입력받은 배열을 이진 트리로 만드는 함수
    private static TreeNode buildTree(String[] nodes) {
        if (nodes.length == 0 || nodes[0].equals("null")) return null;
        TreeNode root = new TreeNode(Integer.parseInt(nodes[0]));
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        int i = 1;
        while (!queue.isEmpty() && i < nodes.length) {
            TreeNode curr = queue.poll();
            if (!nodes[i].equals("null")) {
                curr.left = new TreeNode(Integer.parseInt(nodes[i]));
                queue.add(curr.left);
            }
            i++;
            if (i < nodes.length && !nodes[i].equals("null")) {
                curr.right = new TreeNode(Integer.parseInt(nodes[i]));
                queue.add(curr.right);
            }
            i++;
        }
        return root;
    }
}
```

## 4. 핵심 포인트 및 주의 사항
- **음수 노드 처리**: 모든 노드 값이 음수인 경우, `maxSum`을 0이 아닌 가장 큰 음수 값 하나로 잡아야 합니다. 따라서 `max(0, gain)` 처리는 자식에게만 적용하고, 전역 최댓값 초기값은 `INT_MIN`으로 설정해야 합니다.
- **경로의 정의**: 경로가 꼭 루트를 거칠 필요가 없으므로, 모든 노드에서 '내가 꼭짓점이 되는 경로'를 다 계산해봐야 합니다.
- **재귀 반환값**: 재귀 함수가 반환하는 값은 '나를 포함해서 부모로 이어질 수 있는 한 줄기 경로'여야 한다는 점을 잊지 마세요.
