# [2023 KAKAO BLIND RECRUITMENT] 개인정보 수집 유효기간 - 해설

## 1. 문제 분석
이 문제는 날짜 계산을 정확하게 수행하여 유효기간이 지난 개인정보를 찾아내는 **구현** 문제입니다.

- **알고리즘 유형**: 구현 (Implementation), 시뮬레이션 (Simulation)
- **핵심 포인트**: 
  - "모든 달은 28일까지 있다"는 파격적인 조건을 활용해야 합니다.
  - 연, 월, 일을 각각 계산하면 복잡해질 수 있으므로, 모든 날짜를 **'총 일수(Total Days)'** 하나로 변환하면 비교가 훨씬 쉬워집니다.
  - 1년 = 12개월 = $12 \times 28 = 336$일

## 2. 단계별 풀이 전략

1. **날짜 변환 함수 만들기**:
   - `(연도 * 12 * 28) + (월 * 28) + 일` 공식을 사용하여 "YYYY.MM.DD" 형태의 문자열을 하나의 정수(총 일수)로 바꿉니다.

2. **약관 정보 정리**:
   - `terms` 배열을 순회하며 약관 종류(A, B, C...)를 키(Key)로, 유효기간(달 수)을 값(Value)으로 하는 맵(Map/Dictionary)에 저장합니다. 이때 달 수에 28을 곱해 '일수' 단위로 저장하면 편리합니다.

3. **개인정보 검사**:
   - `privacies` 배열을 하나씩 확인합니다.
   - 수집 일자를 총 일수로 변환한 뒤, 해당 약관의 유효 일수를 더합니다.
   - 이 결과값(만료일)이 `today`의 총 일수보다 작거나 같다면 유효기간이 지난 것이므로 파기 목록에 추가합니다.
     - *주의: "수집일 + 유효기간"이 되는 날부터 파기해야 하므로, 오늘 날짜가 이 날과 같거나 크다면 파기 대상입니다.*

## 3. 코드 구현

### Python
```python
import sys  # 표준 입출력을 위해 사용합니다.

def to_days(date_str):
    """
    "YYYY.MM.DD" 형태의 날짜를 총 일수로 변환하는 함수입니다.
    모든 달은 28일이므로, (연*12*28) + (월*28) + 일로 계산합니다.
    """
    year, month, day = map(int, date_str.split('.'))
    return (year * 12 * 28) + (month * 28) + day

def solution(today, terms, privacies):
    answer = []
    # 오늘 날짜를 총 일수로 변환합니다.
    today_days = to_days(today)
    
    # 약관 정보를 딕셔너리에 저장합니다. (예: {'A': 6*28, 'B': 12*28})
    term_dict = {}
    for term in terms:
        t_type, t_month = term.split()
        term_dict[t_type] = int(t_month) * 28
        
    # 각 개인정보를 순회하며 만료 여부를 확인합니다.
    for i, privacy in enumerate(privacies):
        p_date, p_type = privacy.split()
        # 수집 일자를 일수로 변환하고 유효기간(일수)을 더합니다.
        expire_days = to_days(p_date) + term_dict[p_type]
        
        # 만료일이 오늘 날짜보다 작거나 같다면 파기 대상입니다.
        if expire_days <= today_days:
            answer.append(i + 1) # 번호는 1번부터 시작하므로 i+1을 저장합니다.
            
    return answer

if __name__ == "__main__":
    # 사용자 입력을 시뮬레이션합니다. (실제 환경에 맞게 입력 형식을 조정할 수 있습니다.)
    today = input("오늘 날짜(YYYY.MM.DD): ")
    terms_input = input("약관 목록(예: A 6, B 12): ").split(', ')
    privacies_input = input("개인정보 목록(예: 2021.05.02 A, 2021.07.01 B): ").split(', ')
    
    result = solution(today, terms_input, privacies_input)
    print(f"파기해야 할 번호: {result}")
```

### C++
```cpp
#include <iostream>  // 표준 입출력을 위한 라이브러리입니다.
#include <vector>    // 가변 배열 사용을 위해 필요합니다.
#include <string>    // 문자열 처리를 위해 필요합니다.
#include <sstream>   // 문자열을 자르기(파싱) 위해 사용합니다.
#include <map>       // 약관 종류별 기간을 저장하기 위한 자료구조입니다.

#define endl "\n"    // 줄 바꿈 성능 향상을 위한 설정입니다.

using namespace std; // 표준 이름을 편하게 사용합니다.

/**
 * [toDays 함수 설명]
 * "YYYY.MM.DD" 문자열을 받아 총 일수로 계산하여 반환합니다.
 */
int toDays(string date) {
    int year = stoi(date.substr(0, 4));  // 앞에서 4글자(연도) 추출
    int month = stoi(date.substr(5, 2)); // 5번째부터 2글자(월) 추출
    int day = stoi(date.substr(8, 2));   // 8번째부터 2글자(일) 추출
    return (year * 12 * 28) + (month * 28) + day;
}

/**
 * [solution 함수 설명]
 * 유효기간이 지난 개인정보 번호들을 찾아 오름차순으로 반환합니다.
 */
vector<int> solution(string today, vector<string> terms, vector<string> privacies) {
    vector<int> answer;
    int todayDays = toDays(today); // 오늘의 총 일수
    
    // 약관 정보를 Map에 저장 (Key: 종류, Value: 유효 일수)
    map<string, int> termMap;
    for (string term : terms) {
        stringstream ss(term);
        string type;
        int month;
        ss >> type >> month; // 공백을 기준으로 종류와 달 수를 나눕니다.
        termMap[type] = month * 28;
    }
    
    // 개인정보를 하나씩 검사
    for (int i = 0; i < privacies.size(); i++) {
        string date = privacies[i].substr(0, 10); // 날짜 부분 추출
        string type = privacies[i].substr(11);    // 약관 종류 부분 추출
        
        // (수집일수 + 유효일수)가 오늘 날짜보다 작거나 같으면 만료됨
        if (toDays(date) + termMap[type] <= todayDays) {
            answer.push_back(i + 1); // 1번부터 시작하므로 i+1 저장
        }
    }
    
    return answer;
}

int main() {
    // 사용자 입력을 시뮬레이션합니다.
    string today;
    cout << "오늘 날짜(YYYY.MM.DD): ";
    getline(cin, today);
    
    string termsStr;
    cout << "약관 목록(예: A 6, B 12): ";
    getline(cin, termsStr);
    
    string privaciesStr;
    cout << "개인정보 목록(예: 2021.05.02 A, 2021.07.01 B): ";
    getline(cin, privaciesStr);
    
    vector<string> terms;
    stringstream ssTerms(termsStr);
    string token;
    while (getline(ssTerms, token, ',')) {
        if (!token.empty() && token[0] == ' ') token.erase(0, 1);
        terms.push_back(token);
    }
    
    vector<string> privacies;
    stringstream ssPrivacies(privaciesStr);
    while (getline(ssPrivacies, token, ',')) {
        if (!token.empty() && token[0] == ' ') token.erase(0, 1);
        privacies.push_back(token);
    }
    
    vector<int> result = solution(today, terms, privacies);
    
    cout << "파기할 번호: ";
    for (int num : result) cout << num << " ";
    cout << endl;
    
    return 0;
}
```

### Java
```java
import java.util.ArrayList; // 가변 리스트를 사용하기 위해 불러옵니다.
import java.util.HashMap;   // 약관 정보를 저장할 Map을 위해 불러옵니다.
import java.util.List;      // 리스트 인터페이스를 사용합니다.
import java.util.Map;       // Map 인터페이스를 사용합니다.
import java.util.Scanner;   // 입력을 받기 위해 불러옵니다.

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        // 사용자 입력을 시뮬레이션합니다.
        System.out.print("오늘 날짜(YYYY.MM.DD): ");
        String today = sc.nextLine();
        
        System.out.print("약관 목록(예: A 6, B 12): ");
        String termsLine = sc.nextLine();
        String[] terms = termsLine.split(", ");
        
        System.out.print("개인정보 목록(예: 2021.05.02 A, 2021.07.01 B): ");
        String privaciesLine = sc.nextLine();
        String[] privacies = privaciesLine.split(", ");
        
        int[] result = solution(today, terms, privacies);
        
        System.out.print("파기할 번호: ");
        for (int num : result) {
            System.out.print(num + " ");
        }
        System.out.println();
        sc.close();
    }

    /**
     * "YYYY.MM.DD" 날짜를 총 일수로 변환하는 보조 함수입니다.
     */
    private static int getDays(String date) {
        String[] parts = date.split("\\."); // 점(.)을 기준으로 자릅니다 (정규식 주의)
        int y = Integer.parseInt(parts[0]);
        int m = Integer.parseInt(parts[1]);
        int d = Integer.parseInt(parts[2]);
        return (y * 12 * 28) + (m * 28) + d;
    }

    public static int[] solution(String today, String[] terms, String[] privacies) {
        int todayTotal = getDays(today);
        
        // 약관 종류별 유효 일수를 Map에 저장합니다.
        Map<String, Integer> termMap = new HashMap<>();
        for (String term : terms) {
            String[] t = term.split(" ");
            termMap.put(t[0], Integer.parseInt(t[1]) * 28);
        }
        
        List<Integer> resultList = new ArrayList<>();
        
        // 개인정보 리스트를 순회하며 검사합니다.
        for (int i = 0; i < privacies.length; i++) {
            String[] p = privacies[i].split(" ");
            int collectDays = getDays(p[0]);
            int validDays = termMap.get(p[1]);
            
            // 만료일(수집일 + 유효기간)이 오늘 날짜보다 작거나 같으면 파기 대상입니다.
            if (collectDays + validDays <= todayTotal) {
                resultList.add(i + 1);
            }
        }
        
        // List를 정수 배열로 변환하여 반환합니다.
        int[] answer = new int[resultList.size()];
        for (int i = 0; i < resultList.size(); i++) {
            answer[i] = resultList.get(i);
        }
        return answer;
    }
}
```

## 4. 핵심 포인트 및 주의 사항
- **총 일수 변환**: 모든 달이 28일이라는 조건 덕분에 `12 * 28`을 1년으로 잡고 계산하는 것이 가장 깔끔한 풀이법입니다. 일반적인 날짜 계산(30일, 31일 등)과는 다르므로 주의하세요.
- **문자열 자르기**: C++의 `substr` 인덱스나 Java의 `split(".")` 정규식 처리 등에 익숙해져야 합니다. (Java에서 점은 `\\.`으로 써야 합니다.)
- **오름차순 반환**: `privacies`를 순서대로 검사하며 결과를 담으면 자동으로 오름차순이 유지됩니다.
- **경계값 확인**: 유효기간이 끝나는 날(만료일) 당일도 오늘 날짜와 같다면 파기 대상인지 여부를 명확히 판단해야 합니다. 문제 예시에서 "2022.01.05부터 파기해야 할"이라고 했으므로, 오늘이 2022.01.05라면 파기 대상입니다. 즉, `만료일 <= 오늘` 조건을 사용합니다.
