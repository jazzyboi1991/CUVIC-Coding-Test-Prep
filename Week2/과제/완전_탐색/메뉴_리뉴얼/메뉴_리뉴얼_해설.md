# [2021 KAKAO BLIND RECRUITMENT] 메뉴 리뉴얼 - 해설

## 1. 문제 분석
이 문제는 손님들의 주문 내역에서 가장 많이 나타나는 메뉴 조합을 찾아내는 문제입니다.

- **알고리즘 유형**: 완전 탐색 (Brute Force), 조합 (Combination), 해시 (Hash)
- **핵심 포인트**: 
  - 각 손님이 주문한 단품 메뉴들 중 특정 개수(`course` 배열의 원소)만큼의 조합을 모두 생성해야 합니다.
  - "XY"와 "YX"는 같은 조합으로 취급해야 하므로, 조합을 만들기 전에 **각 주문 문자열을 미리 오름차순으로 정렬**하는 것이 중요합니다.
  - 각 조합이 나타난 횟수를 기록하고, 그 중 가장 많이 주문된 조합(단, 2번 이상)을 결과에 포함합니다.

## 2. 단계별 풀이 전략

1. **사전 준비**:
   - `orders` 배열의 각 문자열을 사전 순으로 정렬합니다. (예: "CBA" → "ABC")

2. **코스 크기별 탐색**:
   - `course` 배열에 있는 각 숫자(코스 요리의 단품 메뉴 개수)에 대해 반복문을 돌립니다.
   - 모든 손님의 주문(`orders`)에 대해 현재 숫자만큼의 모든 **조합(Combination)**을 생성합니다.

3. **빈도수 계산**:
   - 생성된 모든 조합의 빈도수를 맵(Map)이나 딕셔너리에 저장합니다.
   - 해당 코스 크기에서 가장 많이 주문된 횟수(`max_count`)를 찾습니다.

4. **결과 선정**:
   - `max_count`가 2 이상인 경우, 그 횟수만큼 주문된 모든 조합들을 정답 리스트에 추가합니다.

5. **최종 정렬**:
   - 모든 코스 크기에 대해 선정이 끝난 후, 정답 리스트를 사전 순으로 정렬하여 반환합니다.

## 3. 코드 구현

### Python
```python
import sys  # 표준 입출력을 위해 사용합니다.
from itertools import combinations # 조합을 쉽게 만들기 위해 사용합니다.
from collections import Counter    # 빈도수를 세기 위해 사용합니다.

def solution(orders, course):
    answer = []
    
    for size in course:
        temp = []
        for order in orders:
            # 1. 각 주문에서 size개만큼의 모든 조합을 만듭니다.
            # 단, "BA"와 "AB"를 같게 취급하기 위해 정렬된 상태로 조합을 생성합니다.
            combi = combinations(sorted(order), size)
            temp += combi
            
        # 2. 조합들의 빈도수를 계산합니다.
        counter = Counter(temp)
        
        # 3. 만약 조합이 존재하고, 가장 많이 주문된 횟수가 2번 이상이라면
        if len(counter) != 0 and max(counter.values()) >= 2:
            # 4. 가장 많이 나타난 횟수를 찾습니다.
            max_val = max(counter.values())
            # 5. 그 횟수와 일치하는 조합들을 모두 결과에 담습니다.
            for key, value in counter.items():
                if value == max_val:
                    answer.append("".join(key))
                    
    # 최종 결과를 사전 순으로 정렬합니다.
    return sorted(answer)

if __name__ == "__main__":
    # 사용자 입력을 시뮬레이션합니다.
    orders = ["ABCFG", "AC", "CDE", "ACDE", "BCFG", "ACDEH"]
    course = [2, 3, 4]
    
    result = solution(orders, course)
    print(f"새로운 코스 메뉴: {result}")
```

### C++
```cpp
#include <iostream>  // 표준 입출력을 위한 라이브러리입니다.
#include <vector>    // 가변 배열 사용을 위해 필요합니다.
#include <string>    // 문자열 처리를 위해 필요합니다.
#include <map>       // 빈도수 저장을 위한 자료구조입니다.
#include <algorithm> // sort, max 함수 등을 위해 사용합니다.

#define endl "\n"

using namespace std;

// 특정 주문(order)에서 n개의 메뉴를 고르는 모든 조합을 찾는 재귀 함수입니다.
void getCombination(string order, int n, string current, int start, map<string, int>& counts) {
    // 원하는 개수만큼 채워졌다면 빈도수를 기록합니다.
    if (current.length() == n) {
        counts[current]++;
        return;
    }
    
    for (int i = start; i < order.length(); i++) {
        getCombination(order, n, current + order[i], i + 1, counts);
    }
}

vector<string> solution(vector<string> orders, vector<int> course) {
    vector<string> answer;
    
    // 1. 모든 주문 정보를 미리 정렬합니다.
    for (string &s : orders) sort(s.begin(), s.end());
    
    for (int size : course) {
        map<string, int> counts;
        // 2. 각 주문에서 size개만큼의 조합을 모두 추출합니다.
        for (string order : orders) {
            getCombination(order, size, "", 0, counts);
        }
        
        // 3. 해당 크기에서 가장 많이 주문된 횟수를 찾습니다.
        int maxCount = 0;
        for (auto const& [combi, count] : counts) {
            maxCount = max(maxCount, count);
        }
        
        // 4. 최소 2번 이상 주문되었고 최다 빈도인 메뉴들을 결과에 추가합니다.
        if (maxCount >= 2) {
            for (auto const& [combi, count] : counts) {
                if (count == maxCount) answer.push_back(combi);
            }
        }
    }
    
    // 최종 결과를 사전 순으로 정렬합니다.
    sort(answer.begin(), answer.end());
    return answer;
}

int main() {
    // 테스트용 데이터
    vector<string> o = {"ABCFG", "AC", "CDE", "ACDE", "BCFG", "ACDEH"};
    vector<int> c = {2, 3, 4};
    
    vector<string> res = solution(o, c);
    
    cout << "추천 코스: ";
    for (string s : res) cout << s << " ";
    cout << endl;
    
    return 0;
}
```

### Java
```java
import java.util.*; // List, Map, HashMap 등을 사용하기 위해 불러옵니다.

public class Main {
    public static void main(String[] args) {
        // 테스트용 데이터 실행
        String[] orders = {"ABCFG", "AC", "CDE", "ACDE", "BCFG", "ACDEH"};
        int[] course = {2, 3, 4};
        
        String[] result = solution(orders, course);
        System.out.println("추천 코스 메뉴: " + Arrays.toString(result));
    }

    // 빈도수를 기록할 전역 맵입니다.
    static Map<String, Integer> map;

    public static String[] solution(String[] orders, int[] course) {
        List<String> answer = new ArrayList<>();
        
        // 1. 각 손님의 주문 내역을 사전 순으로 정렬합니다.
        for (int i = 0; i < orders.length; i++) {
            char[] chars = orders[i].toCharArray();
            Arrays.sort(chars);
            orders[i] = String.valueOf(chars);
        }

        for (int size : course) {
            map = new HashMap<>(); // 코스 크기별로 맵 초기화
            
            // 2. 모든 주문에서 size 크기의 조합을 생성하여 빈도수를 셉니다.
            for (String order : orders) {
                if (order.length() >= size) {
                    combination(order, "", 0, size);
                }
            }

            // 3. 가장 많이 주문된 횟수를 찾습니다.
            int maxVal = 0;
            for (int count : map.values()) {
                maxVal = Math.max(maxVal, count);
            }

            // 4. 최소 2명 이상 주문되었고 최다 빈도인 것들을 선별합니다.
            if (maxVal >= 2) {
                for (String key : map.keySet()) {
                    if (map.get(key) == maxVal) {
                        answer.add(key);
                    }
                }
            }
        }

        // 사전 순으로 오름차순 정렬
        Collections.sort(answer);
        return answer.toArray(new String[0]);
    }

    /**
     * [combination 재귀 함수]
     * 백트래킹을 사용하여 모든 조합을 만들어 맵에 저장합니다.
     */
    private static void combination(String order, String current, int start, int size) {
        if (current.length() == size) {
            map.put(current, map.getOrDefault(current, 0) + 1);
            return;
        }

        for (int i = start; i < order.length(); i++) {
            combination(order, current + order.charAt(i), i + 1, size);
        }
    }
}
```

## 4. 핵심 포인트 및 주의 사항
- **조합의 사전 정렬**: "XY"와 "YX"가 같은 조합임을 알기 위해 조합 생성 전에 반드시 각 주문 문자열을 정렬해야 합니다.
- **최대 빈도 처리**: 단순히 가장 많이 주문된 것만 찾는 것이 아니라, 해당 코스 크기에서 **공동 1위**인 모든 메뉴 구성을 정답에 포함해야 함을 잊지 마세요.
- **최소 조건**: 최소 2명 이상의 손님으로부터 주문된 것만 코스 후보에 들어갈 수 있습니다. (`max_val >= 2` 조건)
- **최종 정렬**: 모든 코스 크기 탐색이 끝난 후, 정답 리스트 전체를 다시 한번 사전 순으로 정렬해야 합니다.
